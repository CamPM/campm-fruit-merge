function init() {
    load(); const c=document.getElementById('c'); ctx=c.getContext('2d');
    engine=E.create({gravity:{y:1.2}}); resize();
    M.Events.on(engine, 'collisionStart', e => e.pairs.forEach(p => {
        const a=p.bodyA, b=p.bodyB;
        if(a.plugin?.t!==undefined && a.plugin.t===b.plugin.t && a.plugin.t<11) {
            const t=a.plugin.t, pos={x:(a.position.x+b.position.x)/2, y:(a.position.y+b.position.y)/2};
            C.remove(engine.world, [a,b]); const nf=spawnF(pos.x, pos.y, t+1, false); nf.sq=1.5;
            const now=Date.now(); combo=(now-lastM<1500)?Math.min(5, combo+1):1; lastM=now;
            st.score+=(t+1)*10*combo; st.wallet+=(t+1)*20*combo; if(st.score>st.best)st.best=st.score;
            if(!st.found.includes(t+1)) st.found.push(t+1); if(combo>1) floatT(pos.x, pos.y, combo);
            updUI(); save(); playS(600);
        }
    }));
    setupIn(); loop();
}

function resize() {
    W=window.innerWidth; H=window.innerHeight; c.width=W*devicePixelRatio; c.height=H*devicePixelRatio;
    ctx.setTransform(devicePixelRatio,0,0,devicePixelRatio,0,0);
    C.clear(engine.world, false);
    C.add(engine.world, [
        B.rectangle(W/2, H-120+250, W, 500, {isStatic:true, label:'floor'}),
        B.rectangle(7.5, H/2, 15, H*2, {isStatic:true}),
        B.rectangle(W-7.5, H/2, 15, H*2, {isStatic:true})
    ]);
}

function spawnF(x,y,t,isD) {
    const v=HARDNESS[st.eq.sk];
    const b=B.circle(x,y,SIZES[t],{restitution:v[0], friction:v[1], plugin:{t:t}}); C.add(engine.world, b);
    if(isD){ curIdx=nextIdx; nextIdx=Math.floor(Math.random()*5); canDrop=false; setTimeout(()=>canDrop=true, 500); updUI(); }
    return b;
}

function setupIn() {
    const getX=e=>(e.touches?e.touches[0].clientX:e.clientX), getY=e=>(e.touches?e.touches[0].clientY:e.clientY);
    window.addEventListener('touchstart', e=>{ 
        if(!ac) ac=new (window.AudioContext||window.webkitAudioContext)();
        if(e.target.dataset.t) activeP=e.target.dataset.t; 
    }, {passive:false});
    window.addEventListener('touchmove', e=>{ if(isPaused)return; dragP={x:getX(e), y:getY(e)}; if(!activeP) lastX=Math.max(40,Math.min(W-40,dragP.x)); e.preventDefault(); }, {passive:false});
    window.addEventListener('touchend', e=>{ if(isPaused)return; if(activeP) releaseP(dragP.x, dragP.y); else if(canDrop && !isDead && !e.target.closest('.ui')) spawnF(lastX, 165, curIdx, true); dragP={x:0,y:0}; });
}

function releaseP(x,y) {
    const type=activeP; activeP=null; const p=type==='bomb'?50:type==='wipe'?75:150; if(st.wallet<p) return;
    const bds=C.allBodies(engine.world).filter(b=>b.plugin); let used=false;
    if(type==='bomb'){ bds.forEach(b=>{ if(M.Vector.magnitude(M.Vector.sub(b.position,{x,y}))<80) C.remove(engine.world,b); }); used=true; }
    else { const t=bds.find(b=>M.Bounds.contains(b.bounds,{x,y})); if(t){
        if(type==='wipe') bds.forEach(b=>{ if(b.plugin.t===t.plugin.t){ C.remove(engine.world, b); used=true; } });
        if(type==='evo' && t.plugin.t<11){ const pos=t.position; C.remove(engine.world, t); spawnF(pos.x, pos.y, t.plugin.t+1, false); used=true; }
    }}
    if(used){ st.wallet-=p; updUI(); playS(300); }
}

function loop() {
    if(!isPaused) E.update(engine, 16); ctx.clearRect(0,0,W,H); document.body.style.background=THMS[st.eq.th];
    ctx.fillStyle=BORS[st.eq.bd]; ctx.fillRect(0,0,15,H); ctx.fillRect(W-15,0,15,H);
    ctx.setLineDash([5,5]); ctx.strokeStyle="red"; ctx.beginPath(); ctx.moveTo(0,165); ctx.lineTo(W,165); ctx.stroke(); ctx.setLineDash([]);
    const now=Date.now();
    C.allBodies(engine.world).forEach(b => {
        if(!b.plugin) return;
        if(b.position.y-b.circleRadius<165 && b.speed<0.2){ if(!b.tm) b.tm=now; if(now-b.tm>2000) isDead=true; } else b.tm=null;
        ctx.save(); ctx.translate(b.position.x, b.position.y); ctx.rotate(b.angle);
        if(b.sq>1) b.sq -= 0.05; ctx.scale(b.sq||1, b.sq||1);
        
        // HITBOX LOGIC
        let hb = st.eq.hb; ctx.beginPath(); ctx.arc(0,0,b.circleRadius,0,Math.PI*2);
        if(hb==='tiered') ctx.fillStyle = COLORS[b.plugin.t] + "88";
        else if(hb==='white') ctx.fillStyle = "#ffffffaa";
        else if(hb==='rain') ctx.fillStyle = `hsla(${now/10%360}, 70%, 70%, 0.5)`;
        else ctx.fillStyle = "transparent";

        // TARGETING GLOW
        let dist = M.Vector.magnitude(M.Vector.sub(b.position, dragP));
        if(activeP && (activeP==='bomb'?dist<80:M.Bounds.contains(b.bounds, dragP))){

        } else {
            ctx.lineWidth = st.eq.ot==='Thick'?6:2; ctx.strokeStyle = OUTS[st.eq.ot]||"#333";
        }

    });

    if(isDead){ document.getElementById('game-over').style.display='flex'; document.getElementById('gs').innerText=st.score; }
    requestAnimationFrame(loop);
}

function drawCursor(now) {
    ctx.strokeStyle="gold"; ctx.lineWidth=3; ctx.setLineDash([5,5]);
    if(activeP==='bomb'){ ctx.beginPath(); ctx.arc(dragP.x, dragP.y, 80, 0, Math.PI*2); ctx.stroke(); }
    else {
        const s=30; ctx.beginPath();
        ctx.moveTo(dragP.x-s, dragP.y); ctx.lineTo(dragP.x+s, dragP.y);
        ctx.moveTo(dragP.x, dragP.y-s); ctx.lineTo(dragP.x, dragP.y+s); ctx.stroke();
    }
    ctx.setLineDash([]);
}

function playS(f){ 
    if(st.muted || !ac)return; if(ac.state==='suspended')ac.resume();
    const o=ac.createOscillator(), g=ac.createGain(); o.frequency.setValueAtTime(f,ac.currentTime); 
    o.frequency.exponentialRampToValueAtTime(10,ac.currentTime+0.1); g.gain.setValueAtTime(0.05,ac.currentTime); o.connect(g); g.connect(ac.destination); o.start(); o.stop(ac.currentTime+0.1); 
}
